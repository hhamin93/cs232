/*
kevinhamin uses the legit way of reading/parsing the file in kernel
as Prof. Adams asked us to.
this syscall takes char buffer and integer as parameters 
and copy the name of the current process's username into the buffer  
up to the number of characters num.

*/
SYSCALL_DEFINE2(kevinhamin2, char *, buf, int, num){
	
	//declarations.. and initializations...
	const char * s;
	int size;	
	int fd;
	char buffer[1];	
	//name is the one that will be passed onto buf, so it's size is num.
	char name[num];
	int i;
	int final;
	int strLen;	
	char nBuff[1700];
	int k;
	size = 0;
	strLen = 0;
	i = 0;
	k = 0;
	long myUID;
	int uidNum;
	final = 0;
	bool found;
	found = 0;

	printk("start of the syscall___");

	//store uid of the current process into uidNum;
	uidNum = (current->cred)->uid.val;
	printk("uid current process is ... %d\n", uidNum);
	found = false;	

	//use get_fs()and set_fs()  to handle address space mismatch.
	mm_segment_t old_fs = get_fs();
	set_fs(KERNEL_DS);
	
	//open the file /etc/passwd into file struct using sys_open
	fd = sys_open("/etc/passwd", O_RDONLY, 0);
	printk("before the if \n");

	//if the file exist, run thre following code.
	if (fd >= 0) {
	
		//read in the file into nBuff and get the number of characters in the file.
		while(sys_read(fd,buffer,1) == 1){
			nBuff[size] = buffer[0];
			size++;
		}	
	
		//Debuging code... u can ignore this.	
		printk("the buffer: %s\n", nBuff);	
		for(i = 0 ; i < size ; i++)
			printk("buffer at nBuff[i], %c\n", nBuff[i]);
		i = 0;	
		printk("totalsize of characters in the file..: %d\n", size);
	
		//loop through nBuff, aka the file 
		while ( i < size) {
		
			//char variable to read in uid and later to be used for conversion 
			//into integer.	
			char UIDstr[5];	
		
			//copy the name of the process into name.
			while (nBuff[i] != ':') {
				name[k] = nBuff[i];	
				i++;	
				printk("current name, i and k, %s %d %d\n", name, i, k);	
				k++;
			}
			i++;
			k = 0;
	
			//skip the password section		
			while (nBuff[i] != ':') 
				i++;	
			i++;	

			//copy the uid char version into UIDstr uitil :	
			while (nBuff[i] != ':') {
				UIDstr[k] = nBuff[i];	
				strLen++;	
				printk("current UID in loop: %c\n", nBuff[i]);
				i++;	
				k++;
			}
			k = 0;	
		
			//skip until the newline is read and the file
			//is ready for the next process to be read.
			while( nBuff[i] != '\n')		
				i++;	
			i++;	
		
			//make UIDstr NULL terminated string for conversion into int
			UIDstr[strLen] = '\0';
			strLen = 0;
			
			//put UIDstr into const char * for conversion into int
			s = UIDstr;	
	
			//convert character array into interger.	
			if (kstrtol(s, 10, &myUID) == 0) {
				printk( "uid into number success!: %d\n", myUID); 
			}

			//more debugging... 
			printk( "current copied UID is: %d\n", myUID);
			printk("actual UID is : %d\n", uidNum);
		
			//this is the match found from the passwd file with current UID
			//copy name into buf and break out of the loop.	
			if (uidNum == myUID) {
				found = 1;	
				printk( "match found! exiting... \n");
				for(final = 0; final < num; final++)
					buf[final] = name[final];	
			}
		}
	
		printk("this is the namevalue: %s\n", name);

	//if the file does not exist, return EFAULT and error message.
	} else {
		printk(KERN_INFO "the file does not exist. \n");	
		return -EFAULT;	
	}

	//if the file was there but the uid was not matched, return error msg,
	if(found == 0 )
		return -EFAULT;
	//close the file that we opened.
	sys_close(fd);
	
	set_fs(old_fs);
	return 0;
}
